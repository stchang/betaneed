\begin{itemize}
\item  $\C_1 =$ around everything

\end{itemize}

\subsubsection*{Example 1 (AF style)}
\begin{tikzpicture}
\draw[scale=\tkzsc] (0,3) -- (1,3) -- (1,2) -- (2,2) -- (2,1) -- (3,1) -- (3,0);
\draw[scale=\tkzsc,dotted,thick] (0,3) -- (1,2) -- (2,1) -- (3,0);
\filldraw[scale=\tkzsc,black]
(0,3) circle (1mm)
(1,3) circle (1mm)
(1,2) circle (1mm)
(2,2) circle (1mm)
(2,1) circle (1mm)
(3,1) circle (1mm)
(3,0) circle (1mm);
\draw[scale=\tkzsc]
(0,3) node [anchor=south] {$\val_x$}
(1,3) node [anchor=south] {$\val_y$}
(2,3) node [anchor=south] {$\val_z$}
(3,2) node [anchor=west] {$\lx$}
(3,1) node [anchor=west] {$\ly$}
(3,0) node [anchor=west] {$\lz$};
\end{tikzpicture}
\begin{enumerate}
\item 
$\app{\underline{\lx}\app{\ly\app{\lz{x}}{\val_z}}{\val_y}}
     {\underline{\val_x}}$ \\
$\Av[_2] = \val_x$ \\
$\C_1 = \A_1 = \hole$ \\
$\inhole{\C_2}{\E} = \app{\ly\app{\lz{\hole}}{\e_z}}{\e_y}$
\item 
$\app{\lx\app{\underline{\ly}\app{\lz{y}}{\val_z}}
             {\underline{\val_y}}}{\val_x}$ \\
$\Av[_2] = \val_y$ \\
$\C_1 = \app{\lx\hole}{\val_x}$ \\
$\A_1 = \hole$ \\
$\inhole{\C_2}{\E} = \app{\lz{\hole}}{\e_z}$
\item 
$\app{\lx\app{\ly\app{\underline{\lz}{z}}
                     {\underline{\val_z}}}{\val_y}}{\val_x}$ \\
$\Av[_2] = \val_z$ \\
$\C_1 = \app{\lx\app{\ly{\hole}}{\e_y}}{\e_x}$ \\
$\A_1 = \hole$ \\
$\C_2 = \E = \hole$
\end{enumerate}

\subsubsection*{Example 2 (anti-AF style)}
\begin{tikzpicture}

\draw[scale=\tkzsc]
(0,3) node [anchor=south] {$\val_z$}
(1,3) node [anchor=south] {$\val_y$}
(2,3) node [anchor=south] {$\val_x$}
(3,2) node [anchor=west] {$\lx$}
(3,1) node [anchor=west] {$\ly$}
(3,0) node [anchor=west] {$\lz$};
\end{tikzpicture}
\begin{enumerate}
\item 
%% $\app{ \lx\ly\lz{x} }
%%      { \val_x \, \val_y \, \val_z }$ \\
$\app{ \underline{\lx}\ly\lz{x} }
     { \underline{\val_x} \, \val_y \, \val_z }$ \\
$\Av[_2] = \val_x$ \\
$\C_1 = \ap{\hole}{\ap{\val_y}{\val_z}}$ \\
$\A_1 = \hole$ \\
$\C_2 = \ly\lz\hole$ \\
$\E = \hole$
\item 
$\app{ \lx\underline{\ly}\lz{y} }
     { \val_x \, \underline{\val_y} \, \val_z }$ \\
$\Av[_2] = \val_y$ \\
$\C_1 = \ap{\hole}{\val_z}$ \\
$\A_1 = \ap{\lx\hole}{\val_x}$ \\
$\C_2 = \lz\hole$ \\
$\E = \hole$
\item
$\app{ \lx\ly\underline{\lz}{z} }
     { \val_x \, \val_y \, \underline{\val_z} }$ \\
$\Av[_2] = \val_z$ \\
$\C_1 = \hole$ \\
$\A_1 = \app{\lx\ly\hole}{\ap{\val_x}{\val_y}}$ \\
$\C_2 = E = \hole$ 
\end{enumerate}
